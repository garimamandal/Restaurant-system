def def_full_file_reader():
    file_foods = open('files'+navigator_symbol+'list_foods.fsd', 'r') # Reading Food List
    for i in file_foods: # Line by line reading usinf for loop
        list_foods.append(str(i.strip())) #applying this to make the food menu look presentable 
    file_foods.close()

    file_drinks = open('files'+navigator_symbol+'list_drinks.fsd', 'r') # Reading Drinks List
    for i in file_drinks:
        list_drinks.append(str(i.strip()))
    file_drinks.close()

    file_services = open('files'+navigator_symbol+'list_services.fsd', 'r') # Reading Services
    for i in file_services:
        list_services.append(str(i.strip()))
    file_services.close()

    i = 0
    while i <= (len(list_foods) - 1): #for making the price to be on the left most corner to ook presentable
        if 'Rs' in list_foods[i]:
            list_foods[i] = str(list_foods[i][:list_foods[i].index('Rs') - 1]) + ' ' * (20 - (list_foods[i].index('Rs') - 1)) + str(list_foods[i][list_foods[i].index('Rs'):])
        i += 1

    i = 0
    while i <= (len(list_drinks) - 1):
        if 'Rs' in list_drinks[i]:
            list_drinks[i] = str(list_drinks[i][:list_drinks[i].index('Rs') - 1]) + ' ' * (20 - (list_drinks[i].index('Rs') - 1)) + str(list_drinks[i][list_drinks[i].index('Rs'):])
        i += 1

    i = 0
    while i <= (len(list_services) - 1):
        if 'Rs' in list_services[i]:
            list_services[i] = str(list_services[i][:list_services[i].index('Rs') - 1]) + ' ' * (20 - (list_services[i].index('Rs') - 1)) + str(list_services[i][list_services[i].index('Rs'):])
        i += 1
def_full_file_reader()

def def_file_sorter(): # Applying Sorting to the array to be sorted from A to Z ((AND)) taking out prices from the corresponding indexes
    global list_foods, list_drinks, list_services
    list_foods = sorted(list_foods)
    list_drinks = sorted(list_drinks)
    list_services = sorted(list_services)

    i = 0
    while i < len(list_foods):
        list_item_price[i] = float(list_foods[i][int(list_foods[i].index("Rs") + 3):]) # Taking Out "Rs" + [SPACE] from and cast out the string into an integer
        i += 1

    i = 0
    while i < len(list_drinks):
        list_item_price[40 + i] = float(list_drinks[i][int(list_drinks[i].index("Rs") + 3):]) 
        i += 1

    i = 0
    while i < len(list_services):
        list_item_price[80 + i] = float(list_services[i][int(list_services[i].index("Rs") + 3):]) 
        i += 1
def_file_sorter()
